\begin{frame}{A closer look}
	\includegraphics[width=0.6\textwidth]{H-d.png}
	\begin{textblock*}{0.4\paperwidth}(0.58\paperwidth,0.25\paperheight)
		\centering
		Different isotopes of the same elements have a slightly different transition wavelenght\\
		\vspace{0.05\textheight}
		\only<2>{
		Result of two different contributions from the nucleus:\\
		\begin{itemize}
			\item Different mass (Mass Shift)
			\item Different size (Field Shift)
		\end{itemize}
		}
	\end{textblock*}
\end{frame}


\begin{frame}{A closer$^2$ look}
	\only<1>{\includegraphics[width=0.6\textwidth]{H-d.png}}
	\only<2>{\hspace{0.1\textwidth}\includegraphics[width=0.4\textwidth]{shapes.png}}
	\begin{textblock*}{0.4\paperwidth}(0.58\paperwidth,0.25\paperheight)
		\centering
		Some isotopes present a further level splitting:\\
		\vspace{0.05\textheight}
		\textbf{HyperFine Structure (HFS)}\\
		\only<2>{
		Result again of two different contributions from the nucleus:\\
		\begin{itemize}
			\item electrons interaction with magnetic moment of nucleus (Magnetic dipole moment)
			\item electrons interaction with electric moment of nucleus (Electric quadrupole moent)
		\end{itemize}
		}
	\end{textblock*}
\end{frame}
